CREATE DATABASE alimentos_iot;
USE alimentos_iot;

2. Tabla Hogares
CREATE TABLE hogares(
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL,
    direccion VARCHAR(255)
);

Datos iniciales:
INSERT INTO hogares(nombre,direccion)

('Hogar B','Arequipa'),
('Hogar C','Trujillo');

3. Tabla Alimentos
CREATE TABLE alimentos(
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL
);

Datos iniciales:
INSERT INTO alimentos(nombre)
VALUES
('Platano'),
('Tomate'),
('Lechuga'),
('Pollo');

4. Tabla Lecturas
CREATE TABLE lecturas(
    id INT AUTO_INCREMENT PRIMARY KEY,

    hogar_id INT NOT NULL,

    alimento VARCHAR(100),

    estado VARCHAR(50),

    temperatura DECIMAL(5,2),

    humedad DECIMAL(5,2),

    vida_util DECIMAL(5,2),

    fecha DATETIME DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (hogar_id)
    REFERENCES hogares(id)
);
